If you're using Kotlin for your Gradle files:

```groovy
plugins {
    id("com.squareup.sqldelight") version "{{ versions.sqldelight }}"
}

apply(plugin = "com.squareup.sqldelight")

repositories {
    google()
    mavenCentral()
}

sqldelight {
    database("Database") {
        packageName = "com.example"
    }
}
```
